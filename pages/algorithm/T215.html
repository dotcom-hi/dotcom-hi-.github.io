# 215. 数组中的第K个最大元素

# 给定整数数组 nums 和整数 k，请返回数组中第 k 个最大的元素。

# 请注意，你需要找的是数组排序后的第 k 个最大的元素，而不是第 k 个不同的元素。

# input : [3,2,1,5,6,4] 和 k = 2

# out : 5

function quick_sort(nums,l,r,k){
    if(l>=r) return nums[l];
    let i = l-1, j=r+1, x = nums[parseInt((l+r)/2)];  
    // 如果使用 x = nums[Math.floor((l+r)/2)] 会报错 栈溢出
    // 原因不明
    while(i<j){
        do i++; while (nums[i]>x);
        do j--; while (nums[j]<x);
        if(i<j) {
            [nums[i],nums[j]] = [nums[j],nums[i]];
        }
    }
    if(j-l+1>=k) return quick_sort(nums,l,j,k);
    return quick_sort(nums,j+1,r,k-(j-l+1));
}


var findKthLargest = function(nums, k) {
    let n = nums.length;
    let ans = quick_sort(nums,0,n-1,k);
    return ans;    
};
