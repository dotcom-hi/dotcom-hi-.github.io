# 防抖

# 为什么

前端中发生频繁的事件触发：

  • window 的 resize、scroll
  • mousedown、mousemove
  • keyup、keydown

当触发的是回调函数或者Ajex请求的话，会有明显的卡顿出现

解决方案：

  • 防抖 debounce
  • 节流 throttle

# 是什么

原理： 触发事件后，并不即时执行，而是等待n秒。 如果一个事件触发的 n 秒内又触发了这个事件，那就以新的事件的时间为准，刷新n秒倒计时，n 秒后才执行

# 怎么做

第一版，解决 this 在回调函数中指向 window 的问题

function debounce(func, wait) {
    var timeout;
    return function () {
        var context = this; 
        clearTimeout(timeout)
        timeout = setTimeout(function(){func.apply(context)}, wait); // 等待wait秒后，执行func事件。timeout类似指针，指向该定时器。
    }
}

第三版，需求改为：立刻执行函数，然后等到停止触发 n 秒后，才可以重新触发执行

function debounce(func, wait, immediate) {
    var timeout;

    return function () {
        var context = this;
        var args = arguments;

        if (timeout) clearTimeout(timeout);    // 如果之前触发过事件，产生过计时器，立即取消计时器  (如果定时器已经结束，clearTimeout多次也无所谓) 
        
        
        if (immediate) {                       // immediate 为 Boolean，表示是否打开 立即执行 这一需求  
            var callNow = !timeout;            //  timeout为null[即等待wait秒]时，callNow为true，即可执行事件
            timeout = setTimeout(function(){timeout = null;}, wait) // 等待wait秒后 将 timeout 变为 null
            if (callNow) func.apply(context, args)   // 执行事件
        }
        else {
            timeout = setTimeout(function(){func.apply(context, args)}, wait);
        }
    }
}

第四版，函数可能是有返回值

function debounce(func, wait, immediate) {
    var timeout, result;

    return function () {
        var context = this;
        var args = arguments;

        if (timeout) clearTimeout(timeout);
        // 如果已经执行过，不再执行
        if (immediate) {            
            var callNow = !timeout;
            timeout = setTimeout(function(){timeout = null;}, wait)
            if (callNow) result = func.apply(context, args)
        }
        else {
            timeout = setTimeout(function(){func.apply(context, args)}, wait);
        }
        return result;
    }
}

第五版，增加取消功能

function debounce(func, wait, immediate) {
    var timeout, result;

    var debounced = function () {
        var context = this;
        var args = arguments;

        if (timeout) clearTimeout(timeout);
        // 如果已经执行过，不再执行
        if (immediate) {            
            var callNow = !timeout;
            timeout = setTimeout(function(){timeout = null;}, wait)
            if (callNow) result = func.apply(context, args)
        }
        else {
            timeout = setTimeout(function(){func.apply(context, args)}, wait);
        }
        return result;
    }
    debounced.cancel = function(){
        clearTimeout(timeout);
        timeout = null;
    }
    return debounced;
}

# 应用场景

  • 搜索框搜索输入。只需用户最后一次输入完，再发送请求
  • 手机号、邮箱验证输入检测
  • 窗口大小resize。只需窗口调整完成后，计算窗口大小。防止重复渲染。
